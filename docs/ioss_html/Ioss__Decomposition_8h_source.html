165  // T/F if node with global index node owned by this processors ioss-decomp.
166  bool i_own_node(size_t global_index) const
167  {
168  // global_index is 1-based index into global list of nodes [1..global_node_count]
169  return std::binary_search(nodeGTL.begin(), nodeGTL.end(), global_index);
170  }
171 
172  // T/F if element with global index elem owned by this processors ioss-decomp.
173  bool i_own_elem(size_t global_index) const
174  {
175  // global_index is 1-based index into global list of elements [1..global_element_count]
176  return elemGTL.count(global_index) != 0;
177  }
178 
179  size_t node_global_to_local(size_t global_index) const
180  {
181  // global_index is 1-based index into global list of nodes [1..global_node_count]
182  // return value is 1-based index into local list of nodes on this
183  // processor (ioss-decomposition)
184  // Note that for 'int', equivalence and equality are the same, so
185  // lower_bound is OK here (EffectiveSTL, Item 19)
186  typename std::vector<INT>::const_iterator I =
187  lower_bound(nodeGTL.begin(), nodeGTL.end(), global_index);
188  assert(I != nodeGTL.end());
189  return std::distance(nodeGTL.begin(), I) + 1; // Convert to 1-based index.
190  }
191 
192  size_t elem_global_to_local(size_t global_index) const
193  {
194  // global_index is 1-based index into global list of elements [1..global_node_count]
195  // return value is 1-based index into local list of elements on this
196  // processor (ioss-decomposition)
197  typename std::map<INT, INT>::const_iterator I = elemGTL.find(global_index);
198  assert(I != elemGTL.end());
199  return I->second;
200  }

294  private:
295  // This processor "manages" the elements on the exodus mesh file from
296  // element_offset to element_offset+count (0-based). This is
297  // 'file' data
298  //
299  // This processor also appears to the Ioss clients to own other
300  // element and node data based on the decomposition. This is the
301  // 'ioss' data.
302  //
303  // The indices in 'local_element_map' are the elements that are
304  // common to both the 'file' data and the 'ioss' data.
305  // local_element_map[i] contains the location in 'file' data for
306  // the 'ioss' data at location 'i+import_pre_local_elem_index'
307  //
308  // local_element_map[i]+m_elementOffset is the 0-based global index
309  //
310  // The indices in 'import_element_map' map the data received via
311  // mpi communication from other processors into 'ioss' data.
312  // if 'ind=import_element_map[i]', then ioss[ind] = comm_recv[i]
313  // Note that this is the reverse direction of the
314  // local_element_map mapping.
315  //
316  // The indices in 'export_element_map' are used to pull from
317  // 'file' data into the comm_send vector. if 'ind =
318  // export_element_map[i]', then 'comm_send[i] = file[ind]' for i =
319  // 0..#exported_elements
320  //
321  // local_element_map.size() + import_element_map.size() == #
322  // ioss elements on this processor.
323  //
324  // local_element_map.size() + export_element_map.size() == #
325  // file elements on this processor.
326  //
327  // export_element_map and import_element_map are sorted.
328  // The primary key is processor order followed by global id.
329  // The processor association is via 'export_proc_disp' and
330  // 'import_proc_disp' Both are of size '#processors+1' and
331  // the elements for processor p range from [X_proc_disp[p] to
332  // X_proc_disp[p+1])
