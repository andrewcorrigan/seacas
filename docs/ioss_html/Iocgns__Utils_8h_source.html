93  template <typename INT>
94  static void map_cgns_face_to_ioss(const Ioss::ElementTopology *parent_topo, size_t num_to_get,
95  INT *idata)
96  {
97  // The {topo}_map[] arrays map from CGNS face# to IOSS face#.
98  // See http://cgns.github.io/CGNS_docs_current/sids/conv.html#unstructgrid
99  // NOTE: '0' for first entry is to account for 1-based face numbering.
100 
101  switch (parent_topo->shape()) {
102  case Ioss::ElementShape::HEX:
103  static int hex_map[] = {0, 5, 1, 2, 3, 4, 6};
104  for (size_t i = 0; i < num_to_get; i++) {
105  idata[2 * i + 1] = hex_map[idata[2 * i + 1]];
106  }
107  break;
108 
109  case Ioss::ElementShape::TET:
110  static int tet_map[] = {0, 4, 1, 2, 3};
111  for (size_t i = 0; i < num_to_get; i++) {
112  idata[2 * i + 1] = tet_map[idata[2 * i + 1]];
113  }
114  break;
115 
116  case Ioss::ElementShape::PYRAMID:
117  static int pyr_map[] = {0, 5, 1, 2, 3, 4};
118  for (size_t i = 0; i < num_to_get; i++) {
119  idata[2 * i + 1] = pyr_map[idata[2 * i + 1]];
120  }
121  break;
122 
123  case Ioss::ElementShape::WEDGE:
124 #if 0
125  static int wed_map[] = {0, 1, 2, 3, 4, 5}; // Same
126  // Not needed -- maps 1 to 1
127  for (size_t i=0; i < num_to_get; i++) {
128  idata[2*i+1] = wed_map[idata[2*i+1]];
129  }
130 #endif
131  break;
132  default:;
133  }
134  }

136  static void map_ioss_face_to_cgns(const Ioss::ElementTopology *parent_topo, size_t num_to_get,
137  std::vector<cgsize_t> &data)
138  {
139  // The {topo}_map[] arrays map from CGNS face# to IOSS face#.
140  // See http://cgns.github.io/CGNS_docs_current/sids/conv.html#unstructgrid
141  // NOTE: '0' for first entry is to account for 1-based face numbering.
142 
143  switch (parent_topo->shape()) {
144  case Ioss::ElementShape::HEX:
145  static int hex_map[] = {0, 2, 3, 4, 5, 1, 6};
146  for (size_t i = 0; i < num_to_get; i++) {
147  data[num_to_get * 2 + i] = hex_map[data[num_to_get * 2 + i]];
148  }
149  break;
150 
151  case Ioss::ElementShape::TET:
152  static int tet_map[] = {0, 2, 3, 4, 1};
153  for (size_t i = 0; i < num_to_get; i++) {
154  data[num_to_get * 2 + i] = tet_map[data[num_to_get * 2 + i]];
155  }
156  break;
157 
158  case Ioss::ElementShape::PYRAMID:
159  static int pyr_map[] = {0, 2, 3, 4, 5, 1};
160  for (size_t i = 0; i < num_to_get; i++) {
161  data[num_to_get * 2 + i] = pyr_map[data[num_to_get * 2 + i]];
162  }
163  break;
164 
165  case Ioss::ElementShape::WEDGE:
166 #if 0
167  static int wed_map[] = {0, 1, 2, 3, 4, 5}; // Same
168  // Not needed -- maps 1 to 1
169  for (size_t i=0; i < num_to_get; i++) {
170  data[num_to_get * 2 + i] = wed_map[data[num_to_get * 2 + i]];
171  }
172 #endif
173  break;
174  default:;
175  }
176  }
