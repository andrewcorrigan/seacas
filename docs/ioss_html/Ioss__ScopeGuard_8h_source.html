53 class ScopeGuardImplBase
54 {
55  ScopeGuardImplBase &operator=(const ScopeGuardImplBase &);
56 
57 protected:
58  ~ScopeGuardImplBase() = default;
59  ScopeGuardImplBase(const ScopeGuardImplBase &other) : dismissed_(other.dismissed_)
60  {
61  other.Dismiss();
62  }
63  template <typename J> static void SafeExecute(J &j)
64  {
65  if (!j.dismissed_) {
66  try {
67  j.Execute();
68  }
69  catch (...) {
70  }
71  }
72  }
73 
74  mutable bool dismissed_;
75 
76 public:
77  ScopeGuardImplBase() : dismissed_(false) {}
78  void Dismiss() const { dismissed_ = true; }
79 };
